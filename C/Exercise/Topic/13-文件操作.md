## 第 6 题（单选题）

```c
int main()
{ 
  long num=0;
  FILE *fp = NULL;
  if((fp=fopen("fname.dat", "r"))==NULL)
  {
    printf("Can’t open the file!");
    exit(0);
  }
  while(fgetc(fp) != EOF)
  { 
    num++;
  }
  printf("num=%d\n",num);
  fclose(fp);
  return 0;
}
```
